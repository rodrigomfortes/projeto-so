# bochsrc.txt - Configuração do emulador Bochs (Capítulo 2.3.5)
# Referência: The Little OS Book - Chapter 2: Running Bochs

# Quantidade de memória (32MB)
megs: 32

# Imagens ROM do BIOS e VGA
# Nota: Os caminhos podem variar dependendo da instalação do Bochs
# Caminhos comuns no Linux:
romimage: file=/usr/share/bochs/BIOS-bochs-latest
vgaromimage: file=/usr/share/bochs/VGABIOS-lgpl-latest

# Configuração do CD-ROM (nossa ISO)
ata0-master: type=cdrom, path=os.iso, status=inserted

# Boot a partir do CD-ROM
boot: cdrom

# Biblioteca de display (SDL2 ou X11)
# Descomente a linha apropriada para seu sistema:
# display_library: sdl2
# display_library: x

# Log de saída
log: bochsout.txt

# Mouse e teclado
mouse: enabled=0
keyboard: keymap=/usr/share/bochs/keymaps/x11-pc-us.map
